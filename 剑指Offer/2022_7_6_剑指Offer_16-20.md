 *     int val;
 *     ListNode *next;

    ListNode* deleteDuplicates(ListNode* head) {
        if(head==nullptr||head->next==nullptr)
        {
            return head;

        ListNode* prev=head;

        {

                prev->next=cur->next;
                cur=cur->next;
            }

            {
                prev=cur;
                cur=cur->next;
            }

    }
